# isNull

Checks if value is null or undefined.

/**
 * @param {*} value The value to check.
 * @returns {boolean} Returns `true` if `value` is `null`, else `false`.
 *
 * undefined == null  // true
 */
const isNull = value => value === null;

const isNull = value => typeof value === "object" && !value;
